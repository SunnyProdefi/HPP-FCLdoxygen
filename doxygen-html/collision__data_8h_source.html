123  Contact(const CollisionGeometry* o1_, const CollisionGeometry* o2_, int b1_,
124  int b2_, const Vec3f& pos_, const Vec3f& normal_, FCL_REAL depth_)

136  Contact(const CollisionGeometry* o1_, const CollisionGeometry* o2_, int b1_,
137  int b2_, const Vec3f& p1, const Vec3f& p2, const Vec3f& normal_,

141  b1(b1_),
142  b2(b2_),
143  normal(normal_),

619  const CollisionGeometry* o2_, int b1_, int b2_, const Vec3f& p1,

621  if (min_distance > distance) {
622  min_distance = distance;
623  o1 = o1_;
624  o2 = o2_;
625  b1 = b1_;
626  b2 = b2_;

637  o1 = other_result.o1;
638  o2 = other_result.o2;
