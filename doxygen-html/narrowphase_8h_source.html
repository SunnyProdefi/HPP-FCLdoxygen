131  template <typename S1, typename S2>
132  void getGJKInitialGuess(const S1& s1, const S2& s2, Vec3f& guess,
133  support_func_guess_t& support_hint) const {
134  switch (gjk_initial_guess) {
135  case GJKInitialGuess::DefaultGuess:
136  guess = Vec3f(1, 0, 0);
137  support_hint.setZero();
138  break;
139  case GJKInitialGuess::CachedGuess:
140  guess = cached_guess;
141  support_hint = support_func_cached_guess;
142  break;
143  case GJKInitialGuess::BoundingVolumeGuess:
144  if (s1.aabb_local.volume() < 0 || s2.aabb_local.volume() < 0) {
145  HPP_FCL_THROW_PRETTY(
146  "computeLocalAABB must have been called on the shapes before "
147  "using "
148  "GJKInitialGuess::BoundingVolumeGuess.",
149  std::logic_error);
150  }
151  guess.noalias() = s1.aabb_local.center() -
152  (minkowski_difference.oR1 * s2.aabb_local.center() +
153  minkowski_difference.ot1);
154  support_hint.setZero();
155  break;
156  default:
157  HPP_FCL_THROW_PRETTY("Wrong initial guess for GJK.", std::logic_error);
158  }
159  // TODO: use gjk_initial_guess instead
160  HPP_FCL_COMPILER_DIAGNOSTIC_PUSH
161  HPP_FCL_COMPILER_DIAGNOSTIC_IGNORED_DEPRECECATED_DECLARATIONS
162  if (enable_cached_guess) {
163  guess = cached_guess;
164  support_hint = support_func_cached_guess;
165  }
166  HPP_FCL_COMPILER_DIAGNOSTIC_POP
167  }

172  template <typename S1, typename S2>
173  bool runGJKAndEPA(
174  const S1& s1, const Transform3f& tf1, const S2& s2,
175  const Transform3f& tf2, FCL_REAL& distance, bool compute_penetration,
176  Vec3f& p1, Vec3f& p2, Vec3f& normal,
177  bool relative_transformation_already_computed = false) const {
178  bool gjk_and_epa_ran_successfully = true;
179 
180  // Reset internal state of GJK algorithm
181  if (relative_transformation_already_computed)
182  minkowski_difference.set(&s1, &s2);
183  else
184  minkowski_difference.set(&s1, &s2, tf1, tf2);
185  gjk.reset(gjk_max_iterations, gjk_tolerance);
186  epa.status = details::EPA::Status::DidNotRun;
187 
188  // Get initial guess for GJK: default, cached or bounding volume guess
189  Vec3f guess;
190  support_func_guess_t support_hint;
191  getGJKInitialGuess(*minkowski_difference.shapes[0],
192  *minkowski_difference.shapes[1], guess, support_hint);
193 
194  gjk.evaluate(minkowski_difference, guess, support_hint);
195  HPP_FCL_COMPILER_DIAGNOSTIC_PUSH
196  HPP_FCL_COMPILER_DIAGNOSTIC_IGNORED_DEPRECECATED_DECLARATIONS
197  if (gjk_initial_guess == GJKInitialGuess::CachedGuess ||
198  enable_cached_guess) {
199  cached_guess = gjk.getGuessFromSimplex();
200  support_func_cached_guess = gjk.support_hint;
201  }
202  HPP_FCL_COMPILER_DIAGNOSTIC_POP
203 
204  const FCL_REAL dummy_precision =
205  3 * std::sqrt(std::numeric_limits<FCL_REAL>::epsilon());
206  HPP_FCL_UNUSED_VARIABLE(dummy_precision);
207  switch (gjk.status) {
208  case details::GJK::DidNotRun:
209  HPP_FCL_ASSERT(false, "GJK did not run. It should have!",
210  std::logic_error);
211  distance = -(std::numeric_limits<FCL_REAL>::max)();
212  p1 = p2 = normal =
213  Vec3f::Constant(std::numeric_limits<FCL_REAL>::quiet_NaN());
214  gjk_and_epa_ran_successfully = false;
215  break;
216  case details::GJK::Failed:
217  //
218  // GJK ran out of iterations.
219  HPP_FCL_LOG_WARNING("GJK ran out of iterations.");
220  GJKNoCollisionExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
221  normal);
222  gjk_and_epa_ran_successfully = false;
223  break;
224  case details::GJK::NoCollisionEarlyStopped:
225  //
226  // Case where GJK early stopped because the distance was found to be
227  // above the `distance_upper_bound`.
228  // The two witness points have no meaning.
229  GJKEarlyStopExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
230  normal);
231  HPP_FCL_ASSERT(distance >= gjk.distance_upper_bound - dummy_precision,
232  "The distance should be bigger than GJK's "
233  "`distance_upper_bound`.",
234  std::logic_error);
235  break;
236  case details::GJK::NoCollision:
237  //
238  // Case where GJK converged and proved that the shapes are not in
239  // collision, i.e their distance is above GJK's tolerance (default
240  // 1e-6).
241  GJKNoCollisionExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
242  normal);
243  HPP_FCL_ASSERT(std::abs((p1 - p2).norm() - distance) <=
244  gjk.getTolerance() + dummy_precision,
245  "The distance found by GJK should coincide with the "
246  "distance between the closest points.",
247  std::logic_error);
248  break;
249  //
250  // Next are the cases where GJK found the shapes to be in collision, i.e.
251  // their distance is below GJK's tolerance (default 1e-6).
252  case details::GJK::CollisionWithPenetrationInformation:
253  GJKCollisionWithPenetrationInfoExtractWitnessPointsAndNormal(
254  tf1, distance, p1, p2, normal);
255  HPP_FCL_ASSERT(distance <= gjk.getTolerance() + dummy_precision,
256  "The distance found by GJK should be negative (or at )"
257  "least below GJK's tolerance.",
258  std::logic_error);
259  break;
260  case details::GJK::Collision:
261  if (!compute_penetration) {
262  // Skip EPA and set the witness points and the normal to nans.
263  GJKCollisionExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
264  normal);
265  } else {
266  //
267  // Case where the shapes are not inflated (box, cylinder, cone,
268  // convex meshes etc.). We need to run the EPA algorithm to find the
269  // witness points, penetration depth and the normal.
270 
271  // Reset EPA algorithm. Potentially allocate memory if
272  // `epa_max_face_num` or `epa_max_vertex_num` are bigger than EPA's
273  // current storage.
274  epa.reset(epa_max_iterations, epa_tolerance);
275 
276  // TODO: understand why EPA's performance is so bad on cylinders and
277  // cones.
278  epa.evaluate(gjk, -guess);
279 
280  switch (epa.status) {
281  //
282  // In the following switch cases, until the "Valid" case,
283  // EPA either ran out of iterations, of faces or of vertices.
284  // The depth, witness points and the normal are still valid,
285  // simply not at the precision of EPA's tolerance.
286  // The flag `HPP_FCL_ENABLE_LOGGING` enables feebdack on these
287  // cases.
288  //
289  // TODO: Remove OutOfFaces and OutOfVertices statuses and simply
290  // compute the upper bound on max faces and max vertices as a
291  // function of the number of iterations.
292  case details::EPA::OutOfFaces:
293  HPP_FCL_LOG_WARNING("EPA ran out of faces.");
294  EPAValidExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
295  normal);
296  gjk_and_epa_ran_successfully = false;
297  break;
298  case details::EPA::OutOfVertices:
299  HPP_FCL_LOG_WARNING("EPA ran out of vertices.");
300  EPAValidExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
301  normal);
302  gjk_and_epa_ran_successfully = false;
303  break;
304  case details::EPA::Failed:
305  HPP_FCL_LOG_WARNING("EPA ran out of iterations.");
306  EPAValidExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
307  normal);
308  gjk_and_epa_ran_successfully = false;
309  break;
310  case details::EPA::Valid:
311  case details::EPA::AccuracyReached:
312  HPP_FCL_ASSERT(
313  -epa.depth <= epa.getTolerance() + dummy_precision,
314  "EPA's penetration distance should be negative (or "
315  "at least below EPA's tolerance).",
316  std::logic_error);
317  EPAValidExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
318  normal);
319  break;
320  case details::EPA::Degenerated:
321  HPP_FCL_LOG_WARNING(
322  "EPA warning: created a polytope with a degenerated face.");
323  EPAValidExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
324  normal);
325  gjk_and_epa_ran_successfully = true;
326  break;
327  case details::EPA::NonConvex:
328  HPP_FCL_LOG_WARNING(
329  "EPA warning: EPA got called onto non-convex shapes.");
330  EPAValidExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
331  normal);
332  gjk_and_epa_ran_successfully = true;
333  break;
334  case details::EPA::InvalidHull:
335  HPP_FCL_LOG_WARNING("EPA warning: created an invalid polytope.");
336  EPAValidExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
337  normal);
338  gjk_and_epa_ran_successfully = true;
339  break;
340  case details::EPA::DidNotRun:
341  HPP_FCL_ASSERT(false, "EPA did not run. It should have!",
342  std::logic_error);
343  HPP_FCL_LOG_ERROR("EPA error: did not run. It should have.");
344  EPAFailedExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
345  normal);
346  gjk_and_epa_ran_successfully = false;
347  break;
348  case details::EPA::FallBack:
349  HPP_FCL_ASSERT(
350  false,
351  "EPA went into fallback mode. It should never do that.",
352  std::logic_error);
353  HPP_FCL_LOG_ERROR("EPA error: FallBack.");
354  EPAFailedExtractWitnessPointsAndNormal(tf1, distance, p1, p2,
355  normal);
356  gjk_and_epa_ran_successfully = false;
357  break;
358  }
359  }
360  break; // End of case details::GJK::Inside
361  }
362  return gjk_and_epa_ran_successfully;
363  }

383  void GJKNoCollisionExtractWitnessPointsAndNormal(const Transform3f& tf1,
384  FCL_REAL& distance,
385  Vec3f& p1, Vec3f& p2,
386  Vec3f& normal) const {
387  // Apart from early stopping, there are two cases where GJK says there is no
388  // collision:
389  // 1. GJK proved the distance is above its tolerance (default 1e-6).
390  // 2. GJK ran out of iterations.
391  // In any case, `gjk.ray`'s norm is bigger than GJK's tolerance and thus
392  // it can safely be normalized.
393  distance = gjk.distance;
394  const FCL_REAL dummy_precision =
395  3 * std::sqrt(std::numeric_limits<FCL_REAL>::epsilon());
396  HPP_FCL_UNUSED_VARIABLE(dummy_precision);
397  HPP_FCL_ASSERT(
398  gjk.ray.norm() > gjk.getTolerance() - dummy_precision,
399  "The norm of GJK's ray should be bigger than GJK's tolerance.",
400  std::logic_error);
401  normal.noalias() = -tf1.getRotation() * gjk.ray;
402  normal.normalize();
403  // TODO: On degenerated case, the closest points may be non-unique.
404  // (i.e. an object face normal is colinear to `gjk.ray`)
405  gjk.getClosestPoints(minkowski_difference, p1, p2);
406  p1 = tf1.transform(p1);
407  p2 = tf1.transform(p2);
408  }

410  void GJKCollisionWithPenetrationInfoExtractWitnessPointsAndNormal(
411  const Transform3f& tf1, FCL_REAL& distance, Vec3f& p1, Vec3f& p2,
412  Vec3f& normal) const {
413  distance = gjk.distance;
414  gjk.getClosestPoints(minkowski_difference, p1, p2);
415  normal.noalias() = tf1.getRotation() * (p1 - p2);
416  normal.normalize();
417  p1 = tf1.transform(p1);
418  p2 = tf1.transform(p2);
419  }

421  void GJKCollisionExtractWitnessPointsAndNormal(const Transform3f& tf1,
422  FCL_REAL& distance, Vec3f& p1,
423  Vec3f& p2,
424  Vec3f& normal) const {

428  HPP_FCL_UNUSED_VARIABLE(dummy_precision);
429  HPP_FCL_ASSERT(gjk.distance <= gjk.getTolerance() + dummy_precision,

431  std::logic_error);
432  distance = gjk.distance;
433  gjk.getClosestPoints(minkowski_difference, p1, p2);

438  void EPAValidExtractWitnessPointsAndNormal(const Transform3f& tf1,
439  FCL_REAL& distance, Vec3f& p1,
440  Vec3f& p2, Vec3f& normal) const {
441  distance = (std::min)(0., -epa.depth);
442  normal.noalias() = tf1.getRotation() * epa.normal;
443  epa.getClosestPoints(minkowski_difference, p1, p2);
444  // The following is very important to understand why EPA can sometimes
445  // return a normal that is not colinear to the vector $p_1 - p_2$ when
446  // working with tolerances like $\epsilon = 10^{-3}$.
447  // It can be resumed with a simple idea:
448  // EPA is an algorithm meant to find the penetration depth and the
449  // normal. It is not meant to find the closest points.
450  // Again, the issue here is **not** the normal, it's $p_1$ and $p_2$.
451  //
452  // More details:
453  // We'll denote $S_1$ and $S_2$ the two shapes, $n$ the normal and $p_1$ and
454  // $p_2$ the witness points. In theory, when EPA converges to $\epsilon =
455  // 0$, the normal and witness points verify the following property (P):
456  // - $p_1 \in \partial \sigma_{S_1}(n)$,
457  // - $p_2 \in \partial \sigma_{S_2}(-n),
458  // where $\sigma_{S_1}$ and $\sigma_{S_2}$ are the support functions of
459  // $S_1$ and $S_2$. The $\partial \sigma(n)$ simply denotes the support set
460  // of the support function in the direction $n$. (Note: I am leaving out the
461  // details of frame choice for the support function, to avoid making the
462  // mathematical notation too heavy.)
463  // --> In practice, EPA converges to $\epsilon > 0$.
464  // On polytopes and the likes, this does not change much and the property
465  // given above is still valid.
466  // --> However, this is very different on curved surfaces, such as
467  // ellipsoids, cylinders, cones, capsules etc. For these shapes, converging
468  // at $\epsilon = 10^{-6}$ or to $\epsilon = 10^{-3}$ does not change the
469  // normal much, but the property (P) given above is no longer valid, which
470  // means that the points $p_1$ and $p_2$ do not necessarily belong to the
471  // support sets in the direction of $n$ and thus $n$ and $p_1 - p_2$ are not
472  // colinear.
473  //
474  // Do not panic! This is fine.
475  // Although the property above is not verified, it's almost verified,
476  // meaning that $p_1$ and $p_2$ belong to support sets in directions that
477  // are very close to $n$.
478  //
479  // Solution to compute better $p_1$ and $p_2$:
480  // We compute the middle points of the current $p_1$ and $p_2$ and we use
481  // the normal and the distance given by EPA to compute the new $p_1$ and
482  // $p_2$.
483  Vec3f p = tf1.transform((p1 + p2) * 0.5);
484  p1 = p - normal * distance * 0.5;
485  p2 = p + normal * distance * 0.5;
486  }

548  void set(const DistanceRequest& request) {
549  // ---------------------
550  // GJK settings
551  gjk_initial_guess = request.gjk_initial_guess;
552  enable_cached_guess = request.enable_cached_gjk_guess;
553  if (gjk_initial_guess == GJKInitialGuess::CachedGuess ||
