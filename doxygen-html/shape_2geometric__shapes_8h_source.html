350  Capsule(const Capsule& other)
351  : ShapeBase(other), radius(other.radius), halfLength(other.halfLength) {}
352

355 
357  FCL_REAL radius;
358 
360  FCL_REAL halfLength;
361 
363  void computeLocalAABB();
364 
366  NODE_TYPE getNodeType() const { return GEOM_CAPSULE; }
367 
368  FCL_REAL computeVolume() const {
369  return boost::math::constants::pi<FCL_REAL>() * radius * radius *
370  ((halfLength * 2) + radius * 4 / 3.0);
371  }
372 
373  Matrix3f computeMomentofInertia() const {
374  FCL_REAL v_cyl = radius * radius * (halfLength * 2) *
375  boost::math::constants::pi<FCL_REAL>();
376  FCL_REAL v_sph = radius * radius * radius *
377  boost::math::constants::pi<FCL_REAL>() * 4 / 3.0;
378 
379  FCL_REAL h2 = halfLength * halfLength;
380  FCL_REAL r2 = radius * radius;
381  FCL_REAL ix = v_cyl * (h2 / 3. + r2 / 4.) +
382  v_sph * (0.4 * r2 + h2 + 0.75 * radius * halfLength);
383  FCL_REAL iz = (0.5 * v_cyl + 0.4 * v_sph) * radius * radius;
384 
385  return (Matrix3f() << ix, 0, 0, 0, ix, 0, 0, 0, iz).finished();
386  }
387 
388  FCL_REAL minInflationValue() const { return -radius; }
389 
396  std::pair<Capsule, Transform3f> inflated(const FCL_REAL value) const {
397  if (value <= minInflationValue())
398  HPP_FCL_THROW_PRETTY(
399  "value (" << value << ") is two small. It should be at least: "
400  << minInflationValue(),
401  std::invalid_argument);
402  return std::make_pair(Capsule(radius + value, 2 * halfLength),
403  Transform3f());
404  }
405

429  }

445 
447  NODE_TYPE getNodeType() const { return GEOM_CONE; }
448 
449  FCL_REAL computeVolume() const {
450  return boost::math::constants::pi<FCL_REAL>() * radius * radius *
451  (halfLength * 2) / 3;
452  }

454  Matrix3f computeMomentofInertia() const {
455  FCL_REAL V = computeVolume();
456  FCL_REAL ix =
457  V * (0.4 * halfLength * halfLength + 3 * radius * radius / 20);
458  FCL_REAL iz = 0.3 * V * radius * radius;
459 
460  return (Matrix3f() << ix, 0, 0, 0, ix, 0, 0, 0, iz).finished();
461  }
462

473  std::pair<Cone, Transform3f> inflated(const FCL_REAL value) const {
474  if (value <= minInflationValue())
475  HPP_FCL_THROW_PRETTY(
476  "value (" << value << ") is two small. It should be at least: "
477  << minInflationValue(),
478  std::invalid_argument);
479 
480  // tan(alpha) = 2*halfLength/radius;
481  const FCL_REAL tan_alpha = 2 * halfLength / radius;
482  const FCL_REAL sin_alpha = tan_alpha / std::sqrt(1 + tan_alpha * tan_alpha);
483  const FCL_REAL top_inflation = value / sin_alpha;
484  const FCL_REAL bottom_inflation = value;
485 
486  const FCL_REAL new_lz = 2 * halfLength + top_inflation + bottom_inflation;
487  const FCL_REAL new_cz = (top_inflation + bottom_inflation) / 2.;
488  const FCL_REAL new_radius = new_lz / tan_alpha;
489 
490  return std::make_pair(Cone(new_radius, new_lz),
491  Transform3f(Vec3f(0., 0., new_cz)));
492  }
493 
494  private:
495  virtual bool isEqual(const CollisionGeometry& _other) const {
496  const Cone* other_ptr = dynamic_cast<const Cone*>(&_other);
497  if (other_ptr == nullptr) return false;
498  const Cone& other = *other_ptr;
499 
500  return radius == other.radius && halfLength == other.halfLength;
501  }
502 
503  public:
504  EIGEN_MAKE_ALIGNED_OPERATOR_NEW
