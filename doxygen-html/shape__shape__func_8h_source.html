51 template <typename T_SH1, typename T_SH2>
52 struct ShapeShapeDistancer {
53  static FCL_REAL run(const CollisionGeometry* o1, const Transform3f& tf1,
54  const CollisionGeometry* o2, const Transform3f& tf2,
55  const GJKSolver* nsolver, const DistanceRequest& request,
56  DistanceResult& result) {
57  if (request.isSatisfied(result)) return result.min_distance;
58  FCL_REAL distance;
59  Vec3f closest_p1, closest_p2, normal;
60  const T_SH1* obj1 = static_cast<const T_SH1*>(o1);
61  const T_SH2* obj2 = static_cast<const T_SH2*>(o2);
62  nsolver->shapeDistance(*obj1, tf1, *obj2, tf2, distance,
63  request.enable_signed_distance, closest_p1,
64  closest_p2, normal);
65  result.update(distance, obj1, obj2, DistanceResult::NONE,
66  DistanceResult::NONE, closest_p1, closest_p2, normal);
67  return distance;
68  }
69 };

81 template <typename T_SH1, typename T_SH2>
82 struct ShapeShapeCollider {
83  static std::size_t run(const CollisionGeometry* o1, const Transform3f& tf1,
84  const CollisionGeometry* o2, const Transform3f& tf2,
85  const GJKSolver* nsolver,
86  const CollisionRequest& request,
87  CollisionResult& result) {
88  if (request.isSatisfied(result)) return result.numContacts();
89 
90  DistanceResult distanceResult;
91  DistanceRequest distanceRequest;
92  // If the security margin is negative, we have to compute the signed
93  // distance. Otherwise comparison against the secrurity margin makes no
94  // sense.
95  distanceRequest.enable_signed_distance =
96  (request.enable_contact || request.security_margin < 0);
97  FCL_REAL distance = ShapeShapeDistance<T_SH1, T_SH2>(
98  o1, tf1, o2, tf2, nsolver, distanceRequest, distanceResult);
99 
100  size_t num_contacts = 0;
101  const Vec3f& p1 = distanceResult.nearest_points[0];
102  const Vec3f& p2 = distanceResult.nearest_points[1];
103  const Vec3f& normal = distanceResult.normal;
104  FCL_REAL distToCollision = distance - request.security_margin;
105 
106  internal::updateDistanceLowerBoundFromLeaf(request, result, distToCollision,
107  p1, p2, normal);
108  if (distToCollision <= request.collision_distance_threshold &&
109  result.numContacts() < request.num_max_contacts) {
110  if (result.numContacts() < request.num_max_contacts) {
111  const Vec3f& p1 = distanceResult.nearest_points[0];
112  const Vec3f& p2 = distanceResult.nearest_points[1];
113 
114  Contact contact(o1, o2, distanceResult.b1, distanceResult.b2, p1, p2,
115  normal, distance);
116 
117  result.addContact(contact);
118  }
119  num_contacts = result.numContacts();
120  }
121 
122  return num_contacts;
123  }
124 };
